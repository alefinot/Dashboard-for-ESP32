// =========================================================================
// CALCOLO BOUNDING BOX FONT OTTIMIZZATO PER ESP32
// =========================================================================
inline void calculateTextBounds(const GFXfont *font, const char *str, int16_t x, int16_t y,
                                int16_t *x1, int16_t *y1, uint16_t *w, uint16_t *h) {
    if (!font || !str || !*str) {
        *x1 = 0; *y1 = 0; *w = 0; *h = 0;
        return;
    }

    int16_t minx = 32767, miny = 32767, maxx = -32768, maxy = -32768;
    int16_t cursor_x = x, cursor_y = y;
    const uint8_t first = font->first;
    const uint8_t last  = font->last;

    char c;
    while ((c = *str++)) {
        if (c == '\n') {
            cursor_x = x;
            cursor_y += (int16_t)font->yAdvance;
        } else if (c != '\r') {
            uint8_t uc = (uint8_t)c;
            if (uc >= first && uc <= last) {
                const GFXglyph *glyph = &(font->glyph[uc - first]);
                const uint8_t gw = glyph->width;
                const uint8_t gh = glyph->height;

                if (gw && gh) {
                    int16_t gx1 = cursor_x + glyph->xOffset;
                    int16_t gy1 = cursor_y + glyph->yOffset;
                    int16_t gx2 = gx1 + gw - 1;
                    int16_t gy2 = gy1 + gh - 1;

                    if (gx1 < minx) minx = gx1;
                    if (gy1 < miny) miny = gy1;
                    if (gx2 > maxx) maxx = gx2;
                    if (gy2 > maxy) maxy = gy2;
                }
                cursor_x += glyph->xAdvance;
            }
        }
    }

    if (maxx >= minx) {
        *x1 = minx - x;
        *w  = maxx - minx + 1;
    } else {
        *x1 = 0; *w = 0;
    }

    if (maxy >= miny) {
        *y1 = miny - y;
        *h  = maxy - miny + 1;
    } else {
        *y1 = 0; *h = 0;
    }
}

        }
        _panel_instance.setBus(&shared_spi_bus);
        setPanel(&_panel_instance);
    }

    void setFont(const GFXfont* f) {
        _currentGfxFont = f;
        lgfx::LGFX_Device::setFont(f);

inline uint16_t blendColorWithBlack(uint16_t color, float alpha) {
    if (alpha <= 0.02f) return 0x0000; // TFT_BLACK
    if (alpha >= 0.98f) return color;
    
    uint8_t r = (color >> 11) & 0x1F;
    uint8_t g = (color >> 5) & 0x3F;
    uint8_t b = color & 0x1F;
    
    r = (uint8_t)((float)r * alpha);
    g = (uint8_t)((float)g * alpha);
    b = (uint8_t)((float)b * alpha);
    
    return (r << 11) | (g << 5) | b;
}

// 1. LINEA ANTIALIAS (Algoritmo di Xiaolin Wu)
template <typename T>
void drawAALine(T &disp, float x0, float y0, float x1, float y1, uint16_t color) {
    auto plot = [&](int x, int y, float c) {
        if (c <= 0.0f) return;
        uint16_t aaColor = blendColorWithBlack(color, c);
        disp.drawPixel(x, y, aaColor);
    };

    bool steep = fabsf(y1 - y0) > fabsf(x1 - x0);
    if (steep) { std::swap(x0, y0); std::swap(x1, y1); }
    if (x0 > x1) { std::swap(x0, x1); std::swap(y0, y1); }

    float dx = x1 - x0;
    float dy = y1 - y0;
    float gradient = (dx == 0.0f) ? 1.0f : dy / dx;

    // Primo punto
    int xend = (int)roundf(x0);
    float yend = y0 + gradient * (xend - x0);
    float xgap = 1.0f - (x0 + 0.5f - floorf(x0 + 0.5f));
    int xpxl1 = xend;
    int ypxl1 = (int)floorf(yend);

    if (steep) {
        plot(ypxl1,     xpxl1, (1.0f - (yend - floorf(yend))) * xgap);
        plot(ypxl1 + 1, xpxl1, (yend - floorf(yend)) * xgap);
    } else {
        plot(xpxl1, ypxl1,     (1.0f - (yend - floorf(yend))) * xgap);
        plot(xpxl1, ypxl1 + 1, (yend - floorf(yend)) * xgap);
    }
    float intery = yend + gradient;

    // Secondo punto
    int xend2 = (int)roundf(x1);
    float yend2 = y1 + gradient * (xend2 - x1);
    float xgap2 = x1 + 0.5f - floorf(x1 + 0.5f);
    int xpxl2 = xend2;
    int ypxl2 = (int)floorf(yend2);

    if (steep) {
        plot(ypxl2,     xpxl2, (1.0f - (yend2 - floorf(yend2))) * xgap2);
        plot(ypxl2 + 1, xpxl2, (yend2 - floorf(yend2)) * xgap2);
    } else {
        plot(xpxl2, ypxl2,     (1.0f - (yend2 - floorf(yend2))) * xgap2);
        plot(xpxl2, ypxl2 + 1, (yend2 - floorf(yend2)) * xgap2);
    }

    // Loop Principale
    if (steep) {
        for (int x = xpxl1 + 1; x <= xpxl2 - 1; x++) {
            plot((int)floorf(intery),     x, 1.0f - (intery - floorf(intery)));
            plot((int)floorf(intery) + 1, x, intery - floorf(intery));
            intery += gradient;
        }
    } else {
        for (int x = xpxl1 + 1; x <= xpxl2 - 1; x++) {
            plot(x, (int)floorf(intery),     1.0f - (intery - floorf(intery)));
            plot(x, (int)floorf(intery) + 1, intery - floorf(intery));
            intery += gradient;
        }
    }
}

// 2. CERCHIO ANTIALIAS (Algoritmo di Wu)
template <typename T>
void drawAACircle(T &disp, int cx, int cy, int r, uint16_t color) {
    if (r <= 0) return;
    int x = r;
    int y = 0;
    
    auto plot8 = [&](int px, int py, float alpha) {
        if (alpha <= 0.0f) return;
        uint16_t c = blendColorWithBlack(color, alpha);
        disp.drawPixel(cx + px, cy + py, c);
        disp.drawPixel(cx - px, cy + py, c);
        disp.drawPixel(cx + px, cy - py, c);
        disp.drawPixel(cx - px, cy - py, c);
        disp.drawPixel(cx + py, cy + px, c);
        disp.drawPixel(cx - py, cy + px, c);
        disp.drawPixel(cx + py, cy - px, c);
        disp.drawPixel(cx - py, cy - px, c);
    };

    plot8(r, 0, 1.0f);

    float T_val = 0.0f;
    while (x > y) {
        y++;
        float x_exact = sqrtf((float)(r * r - y * y));
        x = (int)ceilf(x_exact);
        T_val = (float)x - x_exact;
        
        plot8(x, y, 1.0f - T_val);
        plot8(x - 1, y, T_val);
    }
}

// 3. RETTANGOLO ARROTONDATO (BADGE) ANTIALIAS
template <typename T>
void drawAACornerArc(T &disp, int cx, int cy, int r, uint8_t corner, uint16_t color) {
    if (r <= 0) return;
    int x = r;
    int y = 0;
    
    int signX = (corner == 0 || corner == 1) ? 1 : -1;
    int signY = (corner == 1 || corner == 2) ? 1 : -1;

    auto plotCorner = [&](int px, int py, float alpha) {
        if (alpha <= 0.0f) return;
        uint16_t c = blendColorWithBlack(color, alpha);
        disp.drawPixel(cx + signX * px, cy + signY * py, c);
        if (px != py) {
            disp.drawPixel(cx + signX * py, cy + signY * px, c);
        }
    };

    float T_val = 0.0f;
    while (x >= y) {
        float x_exact = sqrtf((float)(r * r - y * y));
        x = (int)ceilf(x_exact);
        T_val = (float)x - x_exact;
        
        plotCorner(x, y, 1.0f - T_val);
        plotCorner(x - 1, y, T_val);
        y++;
    }
}

template <typename T>
void drawAARoundRect(T &disp, int x, int y, int w, int h, int r, uint16_t color) {
    if (w <= 0 || h <= 0) return;
    if (r <= 0) {
        drawAALine(disp, x, y, x + w - 1, y, color);
        drawAALine(disp, x + w - 1, y, x + w - 1, y + h - 1, color);
        drawAALine(disp, x + w - 1, y + h - 1, x, y + h - 1, color);
        drawAALine(disp, x, y + h - 1, x, y, color);
        return;
    }
    
    if (r > w / 2) r = w / 2;
    if (r > h / 2) r = h / 2;

    // Lati dritti
    disp.drawFastHLine(x + r, y, w - 2 * r, color);
    disp.drawFastHLine(x + r, y + h - 1, w - 2 * r, color);
    disp.drawFastVLine(x, y + r, h - 2 * r, color);
    disp.drawFastVLine(x + w - 1, y + r, h - 2 * r, color);

    // 4 Angoli in Antialiasing
    drawAACornerArc(disp, x + w - 1 - r, y + r, r, 0, color);         // Top-Right
    drawAACornerArc(disp, x + w - 1 - r, y + h - 1 - r, r, 1, color);     // Bottom-Right
    drawAACornerArc(disp, x + r, y + h - 1 - r, r, 2, color);         // Bottom-Left
    drawAACornerArc(disp, x + r, y + r, r, 3, color);                 // Top-Left
}

void IRAM_ATTR hallSensorISR() {
    constexpr unsigned long DEBOUNCE_US = 10000;
    unsigned long now = micros();
    if (now - lastHallPulseTimeUs > DEBOUNCE_US) { 
        portENTER_CRITICAL_ISR(&hallMux);
        hallPulseIntervalUs = now - lastHallPulseTimeUs;
        lastHallPulseTimeUs = now;
        hallPulseCount++;
        portEXIT_CRITICAL_ISR(&hallMux);
    }
}

inline float getHallSpeed() {
    unsigned long lastTimeUs;
    unsigned long intervalUs;

    portENTER_CRITICAL(&hallMux);
    lastTimeUs = lastHallPulseTimeUs;
    intervalUs = hallPulseIntervalUs;
    portEXIT_CRITICAL(&hallMux);

    if (micros() - lastTimeUs > 2000000UL || intervalUs == 0) {
        return 0.0f;
    }

    return WHEEL_SPEED_FACTOR / (float)intervalUs;
}

void updateFilteredSpeed() {
    float hallSpeed = getHallSpeed();
    int sats = gps.satellites.value();
    bool isGpsValid = gps.speed.isValid() && (sats >= MIN_SATELLITES);

    if (hallSpeed == 0.0f) {
        currentCachedSpeed = 0.0f;
        return;
    }

    if (!isGpsValid) {
        currentCachedSpeed = (hallSpeed < MIN_SPEED_THRESHOLD) ? 0.0f : hallSpeed;
        return;
    }

    float gpsSpeed = (float)gps.speed.kmph();

    if (sats >= OPTIMAL_SATELLITES) {
        currentCachedSpeed = (gpsSpeed < MIN_SPEED_THRESHOLD) ? 0.0f : gpsSpeed;
        return;
    }

    float delta = fabsf(gpsSpeed - hallSpeed);

    if (delta > MAX_SPEED_DELTA_KMH) {
        currentCachedSpeed = (hallSpeed < MIN_SPEED_THRESHOLD) ? 0.0f : hallSpeed;
        return;
    }

    float gpsWeight = (float)(sats - MIN_SATELLITES + 1) / (float)(OPTIMAL_SATELLITES - MIN_SATELLITES + 1);
    float fusedSpeed = (gpsSpeed * gpsWeight) + (hallSpeed * (1.0f - gpsWeight));

    currentCachedSpeed = (fusedSpeed < MIN_SPEED_THRESHOLD) ? 0.0f : fusedSpeed;
}

inline float getFilteredSpeed() {
    return currentCachedSpeed;
}

inline int getDayOfWeek(int y, int m, int d) {
    static const int t[] = { 0, 3, 2, 5, 0, 3, 5, 1, 4, 6, 2, 4 };
    if (m < 3) y -= 1;
    return (y + y/4 - y/100 + y/400 + t[m-1] + d) % 7;
}

int getEuropeanOffset(int year, int month, int day, int hour) {
    if (month < 3 || month > 10) return 1;
    if (month > 3 && month < 10) return 2; 
    int lastSunday = 31 - getDayOfWeek(year, month, 31);
    if (month == 3) { 
        if (day > lastSunday) return 2;
        if (day < lastSunday) return 1;
        return (hour >= 1) ? 2 : 1;
    }
    if (month == 10) { 
        if (day > lastSunday) return 1;
        if (day < lastSunday) return 2;
        return (hour >= 1) ? 1 : 2; 
    }
    return 1;
}

inline int getLocalHour() {
    if (!gps.time.isValid()) return 0;
    int h = gps.time.hour() + currentTimezoneOffset;
    if (h >= 24) h -= 24;
    else if (h < 0) h += 24;
    return h;
}

void handleSerialCommands() {
    static char cmdBuf[32];
    static uint8_t cmdIdx = 0;

    while (Serial.available() > 0) {
        char c = (char)Serial.read();
        if (c == '\n' || c == '\r') {
            if (cmdIdx > 0) {
                cmdBuf[cmdIdx] = '\0';
                for (uint8_t i = 0; i < cmdIdx; i++) {
                    cmdBuf[i] = toupper((unsigned char)cmdBuf[i]);
                }
                if (strcmp(cmdBuf, "REBOOT") == 0 || strcmp(cmdBuf, "RESTART") == 0 || strcmp(cmdBuf, "R") == 0) {
                    Serial.println("[LOG] Riavvio forzato da Serial Monitor ricevuto!");
                    showRebootScreen();
                } else if (strcmp(cmdBuf, "FPS") == 0 || strcmp(cmdBuf, "TOGGLE_FPS") == 0) {
                    showFpsCounter = !showFpsCounter;
                    Serial.printf("[LOG] Contatore FPS %s\n", showFpsCounter ? "ATTIVATO" : "DISATTIVATO");
                } else if (strcmp(cmdBuf, "FPS_TL") == 0) { 
                    fpsBigX = 5; fpsBigY = 0;
                    Serial.println("[LOG] Posizione FPS impostata: TOP-LEFT");
                } else if (strcmp(cmdBuf, "FPS_TR") == 0) { 
                    fpsBigX = 210; fpsBigY = 0;
                    Serial.println("[LOG] Posizione FPS impostata: TOP-RIGHT");
                } else if (strcmp(cmdBuf, "FPS_BL") == 0) { 
                    fpsBigX = 5; fpsBigY = 225;
                    Serial.println("[LOG] Posizione FPS impostata: BOTTOM-LEFT");
                } else if (strcmp(cmdBuf, "FPS_BR") == 0) { 
                    fpsBigX = 210; fpsBigY = 225;
                    Serial.println("[LOG] Posizione FPS impostata: BOTTOM-RIGHT");
                } else if (strncmp(cmdBuf, "FPS_BIG ", 8) == 0) {
                    int x, y;
                    if (sscanf(cmdBuf + 8, "%d %d", &x, &y) == 2) {
                        fpsBigX = constrain(x, 0, 215);
                        fpsBigY = constrain(y, 0, 225);
                        Serial.printf("[LOG] FPS Big Display spostato a X=%d, Y=%d\n", fpsBigX, fpsBigY);
                    }
                }
                cmdIdx = 0;
            }
        } else if (cmdIdx < sizeof(cmdBuf) - 1) {
            cmdBuf[cmdIdx++] = c;
        }
    }
}

void drawFpsOverlay() {
    static float lastDrawnFpsBig = -1.0f;
    static float lastDrawnAvgFpsBig = -1.0f;
    static bool lastStateBig = false;

    static int lastBigX = -1, lastBigY = -1;
    static unsigned long lastFpsDrawTime = 0;

    constexpr int BOX_WIDTH = 105;

    unsigned long now = millis();
    if (now - lastFpsDrawTime < 250 && lastStateBig) return;
    lastFpsDrawTime = now;

    if (!showFpsCounter || lastBigX != fpsBigX || lastBigY != fpsBigY) {
        if (lastStateBig && lastBigX >= 0) {
            bigDisplay.startWrite();
            bigDisplay.fillRect(lastBigX, lastBigY, BOX_WIDTH, 12, TFT_BLACK);

            lastDrawnAvgFpsBig = -1.0f;

        lastBigY = fpsBigY;
    }

    if (!showFpsCounter) return;

    char fpsBuf[24];
    snprintf(fpsBuf, sizeof(fpsBuf), "%.1f | AVG %.1f", currentMeasuredFps, currentAverageFps);

    if (fabsf(currentMeasuredFps - lastDrawnFpsBig) >= 0.3f || fabsf(currentAverageFps - lastDrawnAvgFpsBig) >= 0.3f || !lastStateBig) {
        lastDrawnFpsBig = currentMeasuredFps;
        lastDrawnAvgFpsBig = currentAverageFps;
        lastStateBig = true;
        bigDisplay.startWrite();
        bigDisplay.fillRect(fpsBigX, fpsBigY, BOX_WIDTH, 12, TFT_BLACK);
        bigDisplay.setFont(&Conthrax_SemiBold4pt7b);
        bigDisplay.setTextColor(TFT_GREEN);

        bigDisplay.print(fpsBuf);
        bigDisplay.endWrite();
    }
}

// =========================================================================
void sensorTask(void *pvParameters) {
    TickType_t xLastWakeTime = xTaskGetTickCount();
    const TickType_t xFrequency = pdMS_TO_TICKS(20);

    for (;;) {
        while (gpsSerial.available() > 0) {
            gps.encode(gpsSerial.read());
        }
        
        updateFilteredSpeed();
        processFuelSensor();
        processBatterySensor(); 
        processTemperatureSensor(); 
        updateGPSOdometer();
        processFuelConsumption();
        updateAccelTimer();
        
        if (gps.date.isValid() && gps.time.isValid()) {
            currentTimezoneOffset = getEuropeanOffset(gps.date.year(), gps.date.month(), gps.date.day(), gps.time.hour());
        }

        if (xSemaphoreTake(g_stateMutex, pdMS_TO_TICKS(2)) == pdTRUE) {
            g_sensorData.currentSpeed = currentCachedSpeed;
            g_sensorData.fuelLiters = fuelLiters;
            g_sensorData.fuelPercentage = fuelPercentage;
            g_sensorData.batteryVoltage = batteryVoltage;
            g_sensorData.engineTemperature = engineTemperature;
            g_sensorData.satellites = gps.satellites.value();
            g_sensorData.totalDistanceKm = totalDistanceKm;
            g_sensorData.accelResultTime = accelResultTime;
            g_sensorData.accelState = accelState;
            g_sensorData.instantKml = instantKml;
            g_sensorData.averageKml = averageKml;
            
            g_sensorData.timeValid = gps.time.isValid();
            g_sensorData.dateValid = gps.date.isValid();
            g_sensorData.isGpsSpeedValid = gps.speed.isValid() && (g_sensorData.satellites >= MIN_SATELLITES);

            if (g_sensorData.timeValid) {
                g_sensorData.localHour = getLocalHour();
                g_sensorData.minute = gps.time.minute();
            }
            if (g_sensorData.dateValid) {
                g_sensorData.day = gps.date.day();
                g_sensorData.month = gps.date.month();
                g_sensorData.year = gps.date.year() % 100;
            }

            xSemaphoreGive(g_stateMutex);
        }

        vTaskDelayUntil(&xLastWakeTime, xFrequency);
    }
}

void loop() {
    handleSerialCommands(); 

    unsigned long now = millis();

    if (DISPLAY_REFRESH_MS == 0 || (now - lastDisplayUpdate >= DISPLAY_REFRESH_MS)) {
        
        static unsigned long lastFrameTime = 0;
        static float filteredFrameTimeMs = 16.6f;

        unsigned long frameDeltaMs = now - lastFrameTime;

        if (frameDeltaMs > 0) {
            lastFrameTime = now;
            filteredFrameTimeMs = (filteredFrameTimeMs * 0.95f) + ((float)frameDeltaMs * 0.05f);
            
            if (filteredFrameTimeMs > 0.0f) {
                currentMeasuredFps = 1000.0f / filteredFrameTimeMs;
            }
        }

        static unsigned long lastFpsAvgCalcTime = 0;
        if (now - lastFpsAvgCalcTime >= 250) { 
            lastFpsAvgCalcTime = now;
            
            fpsHistory[fpsHistoryIndex] = currentMeasuredFps;
            fpsHistoryIndex = (fpsHistoryIndex + 1) % FPS_AVG_SAMPLES;
            if (fpsHistoryCount < FPS_AVG_SAMPLES) {
                fpsHistoryCount++;
            }

            float fpsSum = 0.0f;
            for (uint8_t i = 0; i < fpsHistoryCount; i++) {
                fpsSum += fpsHistory[i];
            }
            currentAverageFps = fpsSum / (float)fpsHistoryCount;
        }

        lastDisplayUpdate = now;

        SensorSnapshot snap;
        if (xSemaphoreTake(g_stateMutex, pdMS_TO_TICKS(2)) == pdTRUE) {
            snap = g_sensorData;
            xSemaphoreGive(g_stateMutex);
        }

        updateBigDisplay(snap);
        updateSmallDisplay(snap);
        drawFpsOverlay(); 
        checkNightMode(snap);
    }

    static unsigned long lastTelemetryUpdate = 0;
    if (ENABLE_SERIAL_TELEMETRY && (now - lastTelemetryUpdate >= TELEMETRY_REFRESH_MS)) {
        lastTelemetryUpdate = now;
        SensorSnapshot snap;
        if (xSemaphoreTake(g_stateMutex, pdMS_TO_TICKS(2)) == pdTRUE) {
            snap = g_sensorData;
            xSemaphoreGive(g_stateMutex);
        }
        transmitTelemetry(snap);
    }

    if (DISPLAY_REFRESH_MS > 0) {
        vTaskDelay(pdMS_TO_TICKS(1));
    } else {
        yield(); 
    }
}

void processBatterySensor() {
    int rawADC = analogRead(BATTERY_SENSE_PIN);
    float pinVoltage = (float)rawADC * ADC_VOLTS_FACTOR;
    batteryVoltage = (pinVoltage * 5.7f) + 0.2f;
    if (batteryVoltage < 2.0f) batteryVoltage = 0.0f;
}

void processTemperatureSensor() {
    int rawADC = analogRead(TEMP_SENSE_PIN);
    if (rawADC <= 100 || rawADC >= 4000) { 
        engineTemperature = 0.0f;
        return;
    }
    float vOut = (float)rawADC * ADC_VOLTS_FACTOR;
    float resistance = NTC_R_BALANCE * (vOut / (3.3f - vOut));
    float steinhart = logf(resistance / NTC_R_ROOM) / NTC_BETA + NTC_INV_ROOM_KELVIN; 
    engineTemperature = (1.0f / steinhart) - 273.15f; 
}

void updateGPSOdometer() {
    if (ENABLE_POWER_SENSE && digitalRead(POWER_SENSE_PIN) == LOW) {
        preferences.putDouble("odo", totalDistanceKm);
        preferences.end();
        
        analogWrite(BL_BIG_DISPLAY, 0);
        analogWrite(BL_SMALL_DISPLAY, 0);
        digitalWrite(BL_BIG_DISPLAY, LOW);
        digitalWrite(BL_SMALL_DISPLAY, LOW);
        
        Serial.println("[LOG] Mancanza tensione quadro. Deep Sleep attivo.");
        Serial.flush();
        
        esp_sleep_enable_ext0_wakeup((gpio_num_t)POWER_SENSE_PIN, 1);
        esp_deep_sleep_start(); 
    }

    unsigned long pulses = 0;
    portENTER_CRITICAL(&hallMux);
    pulses = hallPulseCount;
    hallPulseCount = 0;
    portEXIT_CRITICAL(&hallMux);

    bool isGpsValid = (gps.location.isValid() && gps.satellites.value() >= MIN_SATELLITES);

    if (!isGpsValid && pulses > 0) {
        double hallDistKm = (double)pulses * WHEEL_DIST_PER_PULSE_KM;
        totalDistanceKm += hallDistKm;
        tripDistanceKm += hallDistKm;
        
        if (totalDistanceKm - lastSavedOdo >= 1.0) {
            preferences.putDouble("odo", totalDistanceKm);
            lastSavedOdo = totalDistanceKm;
            Serial.println("[LOG] Odo salvato periodicamente (+1 KM da Sensore Hall)");
        }
    }

    if (isGpsValid && gps.location.isUpdated()) {
        float currentSpeed = getFilteredSpeed();
        
        if (currentSpeed > 0.0f) { 
            if (hasLastPos) {
                double distanceMeters = TinyGPSPlus::distanceBetween(gps.location.lat(), gps.location.lng(), lastLat, lastLon);
                if (distanceMeters < 500.0) { 
                    double dKm = (distanceMeters / 1000.0);
                    totalDistanceKm += dKm; 
                    tripDistanceKm += dKm; 
                    if (totalDistanceKm - lastSavedOdo >= 1.0) {
                        preferences.putDouble("odo", totalDistanceKm);
                        lastSavedOdo = totalDistanceKm;
                        Serial.println("[LOG] Odo salvato periodicamente (+1 KM da GPS)");
                    }
                }
            }
            lastLat = gps.location.lat(); 
            lastLon = gps.location.lng();
            hasLastPos = true;
        } else {
            lastLat = gps.location.lat(); 
            lastLon = gps.location.lng();
            hasLastPos = true;
        }
    }
}

void processFuelConsumption() {
    if (tripStartFuelLiters < 0.0f && fuelLiters > 0.0f) {
        tripStartFuelLiters = fuelLiters;
    }

    if (tripStartFuelLiters > 0.0f) {
        float consumed = tripStartFuelLiters - fuelLiters;
        if (consumed > 0.0f) {
            tripFuelConsumedLiters = consumed;
        } else if (consumed < -0.5f) {
            tripStartFuelLiters = fuelLiters;
            tripFuelConsumedLiters = 0.0f;
        }
    }

    if (tripDistanceKm > 0.05 && tripFuelConsumedLiters > 0.01f) {
        averageKml = (float)(tripDistanceKm / tripFuelConsumedLiters);
        if (averageKml > 99.9f) averageKml = 99.9f;
    } else {
        averageKml = 0.0f;
    }

    static unsigned long lastInstSampleTime = 0;
    static double lastInstDistKm = 0.0;
    static float lastInstFuelLiters = 0.0f;

    if (millis() - lastInstSampleTime >= 3000) {
        lastInstSampleTime = millis();

        double dDist = tripDistanceKm - lastInstDistKm;
        float dFuel = lastInstFuelLiters - fuelLiters;

        lastInstDistKm = tripDistanceKm;
        lastInstFuelLiters = fuelLiters;

        float currentSpeed = getFilteredSpeed();

        if (currentSpeed > 0.0f && dDist > 0.005 && dFuel > 0.001f) {
            float rawInst = (float)(dDist / dFuel);
            if (rawInst > 99.9f) rawInst = 99.9f;
            instantKml = (rawInst * 0.4f) + (instantKml * 0.6f); 
        } else if (currentSpeed == 0.0f) {
            instantKml = 0.0f;
        }
    }
}

void updateAccelTimer() {
    float currentSpeed = getFilteredSpeed();
    static bool readyForNextRun = false;

    switch (accelState) {
        case READY:
            accelResultTime = 0.0f; 
            if (currentSpeed >= ACCEL_START_SPEED) { 
                accelState = RUNNING; 
                accelStartTime = millis(); 
                accelResultTime = 0.0f; 
                readyForNextRun = false;
            }
            break;

        case RUNNING:
            accelResultTime = (millis() - accelStartTime) / 1000.0f;
            if (currentSpeed >= ACCEL_TARGET_SPEED) { 
                accelState = FINISHED; 
                readyForNextRun = false;
            } else if (currentSpeed < ACCEL_START_SPEED) { 
                accelState = READY; 
                accelResultTime = 0.0f; 
            }
            break;

        case FINISHED:
            if (currentSpeed < ACCEL_START_SPEED) {
                readyForNextRun = true;
            }
            
            if (readyForNextRun && currentSpeed >= ACCEL_START_SPEED) {
                accelState = RUNNING;
                accelStartTime = millis();
                accelResultTime = 0.0f;
                readyForNextRun = false;
            }
            break;
    }
}

void processFuelSensor() {
    int instantReading;

    if (IS_WOKWI_SIMULATOR) {
        instantReading = analogRead(FUEL_TOUCH_PIN); 
    } else {
        instantReading = touchRead(FUEL_TOUCH_PIN); 
    }

    filteredReading = ((float)instantReading * FUEL_FILTER_ALPHA) + (filteredReading * (1.0f - FUEL_FILTER_ALPHA));
    
    if (filteredReading >= touchTable[0]) { fuelLiters = 0.0f; fuelPercentage = 0; return; }
    if (filteredReading <= touchTable[NUM_POINTS - 1]) { fuelLiters = (float)(NUM_POINTS - 1); fuelPercentage = 100; return; }
    
    for (int i = 0; i < NUM_POINTS - 1; i++) {
        int highPoint = touchTable[i];
        int lowPoint = touchTable[i+1];
        if (filteredReading <= highPoint && filteredReading >= lowPoint) {
            float fraction = (filteredReading - highPoint) / (float)(lowPoint - highPoint);
            fuelLiters = (float)i + fraction; 
            fuelPercentage = constrain((int)((fuelLiters / (float)(NUM_POINTS - 1)) * 100.0f), 0, 100);
            return;
        }
    }
}

    int displaySat = isSelfTestActive ? overrideSat : snap.satellites;
    float displayBat = isSelfTestActive ? overrideBat : snap.batteryVoltage;
    float displayTmr = isSelfTestActive ? overrideTimer : snap.accelResultTime;
    TimerState displayAccelState = isSelfTestActive ? RUNNING : snap.accelState;

    int bottomY = BIG_CENTER_Y + OFFSET_BIG_BOTTOM_ROW_Y;
    int bottomCenterX = BIG_CENTER_X + OFFSET_BIG_BOTTOM_ROW_X;

    static int w_sat_max = 0, w_bat_max = 0, w_badge_max = 0, w_tmr_max = 0;
    static uint16_t h_sat_max = 0, h_bat_max = 0, h_tmr_max = 0;
    static uint16_t w_bat_num_max = 0;
    static int totalRowW_fixed = 0;
    static bool bottomRowLayoutInit = false;

    if (!bottomRowLayoutInit) {
        bottomRowLayoutInit = true;
        int16_t bx1, by1; uint16_t bw, bh;
        
        bigDisplay.setFont(&DS_DIGIT15pt7b);
        bigDisplay.getTextBounds("88", 0, 0, &bx1, &by1, &bw, &bh);
        w_sat_max = 16 + 6 + bw;
        h_sat_max = (bh > 15) ? bh : 15;

        bigDisplay.setFont(&DS_DIGIT15pt7b);
        bigDisplay.getTextBounds("88.8", 0, 0, &bx1, &by1, &bw, &bh);
        w_bat_num_max = bw;
        uint16_t w_v_unit, h_v_unit;
        bigDisplay.setFont(&Conthrax_SemiBold7pt7b);
        bigDisplay.getTextBounds(" V", 0, 0, &bx1, &by1, &w_v_unit, &h_v_unit);
        
        uint16_t batIconH = 18; 
        w_bat_max = 10 + 4 + w_bat_num_max + w_v_unit; 
        h_bat_max = (bh > batIconH) ? bh : batIconH;
        if (h_v_unit > h_bat_max) h_bat_max = h_v_unit;

        int16_t bx_b1, by_b1, bx_b2, by_b2;
        uint16_t bw1, bh1, bw2, bh2;
        bigDisplay.setFont(&Conthrax_SemiBold4pt7b);
        bigDisplay.getTextBounds(ACCEL_BADGE_LINE1, 0, 0, &bx_b1, &by_b1, &bw1, &bh1);
        bigDisplay.getTextBounds(ACCEL_BADGE_LINE2, 0, 0, &bx_b2, &by_b2, &bw2, &bh2);
        uint16_t max_text_w = (bw1 > bw2) ? bw1 : bw2;
        w_badge_max = max_text_w + 8; 

        bigDisplay.setFont(&DS_DIGIT15pt7b);
        bigDisplay.getTextBounds("88.88", 0, 0, &bx1, &by1, &bw, &bh);
        uint16_t w_s_unit, h_s_unit;
        bigDisplay.setFont(&Conthrax_SemiBold7pt7b);
        bigDisplay.getTextBounds(" S", 0, 0, &bx1, &by1, &w_s_unit, &h_s_unit);
        w_tmr_max = bw + w_s_unit;
        h_tmr_max = (bh > h_s_unit) ? bh : h_s_unit;

        int gapX = 14;
        int badgeGapX = 5; 
        int batGapX = 4;   
        totalRowW_fixed = w_sat_max + gapX + w_badge_max + badgeGapX + w_tmr_max + batGapX + w_bat_max;
    }

    int gapX = 14;
    int badgeGapX = 5;
    int batGapX = 4; 
    int bottomStartX = bottomCenterX - (totalRowW_fixed / 2);

    int satX   = bottomStartX;
    int badgeX = satX + w_sat_max + gapX;
    int tmrX   = badgeX + w_badge_max + badgeGapX;
    int batX   = tmrX + w_tmr_max + batGapX;

    // --- 1. SATELLITI ---
    if (displaySat != lastSat || forceDraw) {
        lastSat = displaySat;
        componentUpdated = true;

        char satStr[8]; snprintf(satStr, sizeof(satStr), "%d", displaySat);
        bigDisplay.setFont(&DS_DIGIT15pt7b);
        bigDisplay.getTextBounds(satStr, 0, 0, &x1, &y1, &w, &h);

        bigDisplay.fillRect(satX - 1, bottomY - 1, w_sat_max + 2, h_sat_max + 2, TFT_BLACK);
        drawLocationIcon(satX, bottomY + 2, TFT_WHITE); 
        bigDisplay.setFont(&DS_DIGIT15pt7b); bigDisplay.setTextColor(TFT_WHITE);
        bigDisplay.setCursor(satX + 16 + 6 - x1, bottomY - y1); bigDisplay.print(satStr);
        drawDebugBox(bigDisplay, satX - 1, bottomY - 1, w_sat_max + 2, h_sat_max + 2);
    }

    // --- 2. BADGE 0-50 (con Antialiasing) ---
    if (displayAccelState != lastState || forceDraw) {
        lastState = displayAccelState;
        componentUpdated = true;

        uint16_t timerColor = (displayAccelState == RUNNING) ? TFT_YELLOW : ((displayAccelState == FINISHED) ? TFT_GREEN : TFT_WHITE);
        
        int16_t bx1_b, by1_b, bx2_b, by2_b;
        uint16_t bw1_b, bh1_b, bw2_b, bh2_b;
        bigDisplay.setFont(&Conthrax_SemiBold4pt7b); 
        bigDisplay.getTextBounds(ACCEL_BADGE_LINE1, 0, 0, &bx1_b, &by1_b, &bw1_b, &bh1_b);
        bigDisplay.getTextBounds(ACCEL_BADGE_LINE2, 0, 0, &bx2_b, &by2_b, &bw2_b, &bh2_b);
        
        int badgeW = w_badge_max;
        int badgeH = 20; 
        int badgePosY = bottomY - 1;

        bigDisplay.fillRect(badgeX - 1, badgePosY - 1, badgeW + 2, badgeH + 2, TFT_BLACK);
        drawAARoundRect(bigDisplay, badgeX, badgePosY, badgeW, badgeH, 3, timerColor); 
        bigDisplay.setFont(&Conthrax_SemiBold4pt7b); 
        bigDisplay.setTextColor(timerColor);

        bigDisplay.setCursor(badgeX + (badgeW - bw1_b) / 2 - bx1_b, badgePosY + 3 - by1_b); 
        bigDisplay.print(ACCEL_BADGE_LINE1);

        bigDisplay.setCursor(badgeX + (badgeW - bw2_b) / 2 - bx2_b, badgePosY + 3 + bh1_b + 2 - by2_b); 
        bigDisplay.print(ACCEL_BADGE_LINE2);

        drawDebugBox(bigDisplay, badgeX - 1, badgePosY - 1, badgeW + 2, badgeH + 2);
    }

    // --- 3. TIMER ---
    char tmrStr[8]; snprintf(tmrStr, sizeof(tmrStr), "%.2f", displayTmr);
    static char lastTmrStr[8] = "";
    static TimerState lastStateTimer = READY;

    if (strcmp(tmrStr, lastTmrStr) != 0 || displayAccelState != lastStateTimer || forceDraw) {
        strcpy(lastTmrStr, tmrStr);
        lastStateTimer = displayAccelState;
        componentUpdated = true;

        uint16_t timerColor = (displayAccelState == RUNNING) ? TFT_YELLOW : ((displayAccelState == FINISHED) ? TFT_GREEN : TFT_WHITE);
        const char* tmrUnitStr = " S";
        
        bigDisplay.setFont(&DS_DIGIT15pt7b);
        bigDisplay.getTextBounds(tmrStr, 0, 0, &x1, &y1, &w, &h);

        bigDisplay.fillRect(tmrX - 1, bottomY - 1, w_tmr_max + 2, h_tmr_max + 2, TFT_BLACK);
        bigDisplay.setFont(&DS_DIGIT15pt7b); 
        bigDisplay.setCursor(tmrX - x1, bottomY - y1); 
        bigDisplay.setTextColor(timerColor); 
        bigDisplay.print(tmrStr);
        bigDisplay.setFont(&Conthrax_SemiBold7pt7b); 
        bigDisplay.print(tmrUnitStr);

        drawDebugBox(bigDisplay, tmrX - 1, bottomY - 1, w_tmr_max + 2, h_tmr_max + 2);
    }

    // --- 4. BATTERIA ---
    char batStr[8]; snprintf(batStr, sizeof(batStr), "%.1f", displayBat);
    static char lastBatStr[8] = "";

    if (strcmp(batStr, lastBatStr) != 0 || forceDraw) {
        strcpy(lastBatStr, batStr);
        componentUpdated = true;

        uint16_t batColor = TFT_YELLOW;
        if (displayBat > 0.5f && displayBat < 11.8f) batColor = TFT_RED;
        else if (displayBat >= 13.1f) batColor = TFT_GREEN;
        else if (displayBat <= 0.5f) batColor = TFT_WHITE;

        bigDisplay.fillRect(batX - 1, bottomY - 1, w_bat_max + 2, h_bat_max + 3, TFT_BLACK);

        int iconX = batX;
        int iconY = bottomY + 1; 
        drawBatteryIcon(iconX, iconY, displayBat, batColor);

        int textStartX = batX + 10 + 4;

        bigDisplay.setFont(&DS_DIGIT15pt7b); 
        bigDisplay.getTextBounds(batStr, 0, 0, &x1, &y1, &w, &h);
        bigDisplay.setTextColor(batColor);
        bigDisplay.setCursor(textStartX - x1, bottomY - y1); 
        bigDisplay.print(batStr);

        bigDisplay.setFont(&Conthrax_SemiBold7pt7b); 
        bigDisplay.print(" V");

        drawDebugBox(bigDisplay, batX - 1, bottomY - 1, w_bat_max + 2, h_bat_max + 3);
    }

    // Cerchio di Test Esterno in Antialiasing
    if ((componentUpdated || forceDraw) && ENABLE_CIRCLE_TEST) {
        for (int i = 0; i < 4; i++) { drawAACircle(bigDisplay, BIG_CENTER_X, BIG_CENTER_Y, 173 - i, TFT_CYAN); }
    }

    bigDisplay.endWrite();
}

void updateSmallDisplay(const SensorSnapshot& snap) {
    static int lastFuelPct = -1;
    static char lastLitStr[8] = "";
    static int lastTempInt = -999;
    static float lastDispInstKml = -1.0f;
    static float lastDispAvgKml = -1.0f;
    static bool firstRunSmall = true;
    static bool lastSelfTestStateSmall = false;

    smallDisplay.startWrite();

    bool selfTestChangedSmall = (lastSelfTestStateSmall != isSelfTestActive);
    lastSelfTestStateSmall = isSelfTestActive;
    
    bool forceDraw = firstRunSmall || isSelfTestActive || selfTestChangedSmall;
    if (firstRunSmall) firstRunSmall = false;

    int16_t x15, y15, x50, y50; 
    uint16_t w15, h15, w50, h50; 
    
    smallDisplay.setTextSize(1);

    int displayFuelPct = isSelfTestActive ? overrideFuelPct : snap.fuelPercentage;
    float displayFuelLiters = isSelfTestActive ? overrideFuelLiters : snap.fuelLiters;
    float displayInstKml = isSelfTestActive ? overrideInstKml : snap.instantKml;
    float displayAvgKml = isSelfTestActive ? overrideAvgKml : snap.averageKml;
    float displayTemp = isSelfTestActive ? overrideTemp : snap.engineTemperature;

    char litStr[6]; snprintf(litStr, sizeof(litStr), "%.1f", displayFuelLiters);

    if (displayFuelPct != lastFuelPct || strcmp(litStr, lastLitStr) != 0 || forceDraw) {
        lastFuelPct = displayFuelPct;
        strcpy(lastLitStr, litStr);
        
        int fuelPctX = SMALL_CENTER_X + OFFSET_SMALL_FUEL_PCT_X;
        int fuelPctY = SMALL_CENTER_Y + OFFSET_SMALL_FUEL_PCT_Y;
        
        int fuelLitX = SMALL_CENTER_X + OFFSET_SMALL_FUEL_LITERS_X;
        int fuelLitY = SMALL_CENTER_Y + OFFSET_SMALL_FUEL_LITERS_Y;

        static int refPctX = 0;
        static uint16_t w_pct3_max = 0;
        static uint16_t w_pct_unit_max = 0;
        static bool fuelPctLayoutInit = false;

        if (!fuelPctLayoutInit) {
            fuelPctLayoutInit = true;
            int16_t tx1, ty1; uint16_t tw1, th1, tw2, th2;

            smallDisplay.setFont(&DS_DIGIT15pt7b);
            smallDisplay.getTextBounds("00", 0, 0, &tx1, &ty1, &tw1, &th1);
            smallDisplay.setFont(&Conthrax_SemiBold12pt7b);
            smallDisplay.getTextBounds("%", 0, 0, &tx1, &ty1, &tw2, &th2);
            w_pct_unit_max = tw2;

            uint16_t total2W = tw1 + tw2;
            int start2X = fuelPctX - (total2W / 2);
            refPctX = start2X + tw1;

            smallDisplay.setFont(&DS_DIGIT15pt7b);
            smallDisplay.getTextBounds("100", 0, 0, &tx1, &ty1, &w_pct3_max, &th1);
        }

        static int refLitX = 0;
        static uint16_t w_lit_num_max = 0;
        static uint16_t w_lit_unit_max = 0;
        static bool litLayoutInit = false;

        if (!litLayoutInit) {
            litLayoutInit = true;
            int16_t tx1, ty1; uint16_t tw1, th1, tw2, th2;

            smallDisplay.setFont(&DS_DIGIT50pt7b);
            smallDisplay.getTextBounds("9.9", 0, 0, &tx1, &ty1, &tw1, &th1);
            w_lit_num_max = tw1;

            smallDisplay.setFont(&Conthrax_SemiBold12pt7b);
            smallDisplay.getTextBounds("L", 0, 0, &tx1, &ty1, &tw2, &th2);
            w_lit_unit_max = tw2;

            uint16_t totalLitW_max = w_lit_num_max + 6 + tw2;
            int startLitX = fuelLitX - (totalLitW_max / 2);
            refLitX = startLitX + w_lit_num_max; 
        }

        char pctNumStr[6]; snprintf(pctNumStr, sizeof(pctNumStr), "%d", displayFuelPct);
        uint16_t w_pnum, h_pnum; int16_t tx15, ty15;
        smallDisplay.setFont(&DS_DIGIT15pt7b); smallDisplay.getTextBounds(pctNumStr, 0, 0, &tx15, &ty15, &w_pnum, &h_pnum);

        smallDisplay.setFont(&DS_DIGIT50pt7b); smallDisplay.getTextBounds(litStr, 0, 0, &x50, &y50, &w50, &h50);

        int fuelBaselineY = fuelLitY - y50 - (h50 / 2);

        int clearPctLeft = refPctX - w_pct3_max - 4;
        int clearPctWidth = w_pct3_max + w_pct_unit_max + 8;
        smallDisplay.fillRect(clearPctLeft, fuelPctY - 2, clearPctWidth, h_pnum + 4, TFT_BLACK); 

        int clearLitLeft = refLitX - w_lit_num_max - 4;
        int clearLitWidth = w_lit_num_max + 6 + w_lit_unit_max + 8;
        smallDisplay.fillRect(clearLitLeft, fuelLitY - (h50 / 2) - 4, clearLitWidth, h50 + 8, TFT_BLACK); 

        // --- ARCO CARBURANTE (in Antialiasing a passo 0.5°) ---
        int targetFuelIdx = (displayFuelPct * (NUM_ARC_STEPS - 1)) / 100;
        static int lastFuelDrawnIdx = -1;

        if (forceDraw || lastFuelDrawnIdx < 0) {
            for (int i = 0; i < NUM_ARC_STEPS; i++) {
                float a = fuelArcLUT[i].angle;
                float p = (a - 210.0f) / 120.0f; 
                uint16_t segmentColor;
                if (i <= targetFuelIdx) {
                    uint8_t r_comp = (p < 0.5f) ? 31 : (31 - (uint8_t)((p - 0.5f) * 62.0f));
                    uint8_t g_comp = (p > 0.5f) ? 63 : (uint8_t)(p * 126.0f);
                    segmentColor = (r_comp << 11) | (g_comp << 5);
                } else { 
                    segmentColor = smallDisplay.color565(35, 35, 35); 
                }
                drawAALine(smallDisplay, fuelArcLUT[i].xStart, fuelArcLUT[i].yStart, fuelArcLUT[i].xEnd, fuelArcLUT[i].yEnd, segmentColor);
            }
            lastFuelDrawnIdx = targetFuelIdx;
        } else if (targetFuelIdx != lastFuelDrawnIdx) {
            if (targetFuelIdx > lastFuelDrawnIdx) {
                for (int i = lastFuelDrawnIdx + 1; i <= targetFuelIdx; i++) {
                    float a = fuelArcLUT[i].angle;
                    float p = (a - 210.0f) / 120.0f;
                    uint8_t r_comp = (p < 0.5f) ? 31 : (31 - (uint8_t)((p - 0.5f) * 62.0f));
                    uint8_t g_comp = (p > 0.5f) ? 63 : (uint8_t)(p * 126.0f);
                    uint16_t segmentColor = (r_comp << 11) | (g_comp << 5);
                    drawAALine(smallDisplay, fuelArcLUT[i].xStart, fuelArcLUT[i].yStart, fuelArcLUT[i].xEnd, fuelArcLUT[i].yEnd, segmentColor);
                }
            } else {
                uint16_t offColor = smallDisplay.color565(35, 35, 35);
                for (int i = targetFuelIdx + 1; i <= lastFuelDrawnIdx; i++) {
                    drawAALine(smallDisplay, fuelArcLUT[i].xStart, fuelArcLUT[i].yStart, fuelArcLUT[i].xEnd, fuelArcLUT[i].yEnd, offColor);
                }
            }
            lastFuelDrawnIdx = targetFuelIdx;
        }

        int pctBaselineY = fuelPctY - ty15; 
        smallDisplay.setFont(&DS_DIGIT15pt7b); 
        smallDisplay.setCursor(refPctX - w_pnum - tx15, pctBaselineY);
        smallDisplay.setTextColor(TFT_WHITE); 
        smallDisplay.print(pctNumStr);

        smallDisplay.setFont(&Conthrax_SemiBold12pt7b); 
        smallDisplay.setCursor(refPctX, pctBaselineY);
        smallDisplay.print("%");

        drawDebugBox(smallDisplay, refPctX - w_pnum - tx15 - 2, pctBaselineY + ty15 - 2, w_pnum + w_pct_unit_max + 6, h_pnum + 4);

        smallDisplay.setFont(&DS_DIGIT50pt7b); 
        smallDisplay.setCursor(refLitX - w50 - x50, fuelBaselineY);
        smallDisplay.setTextColor(TFT_WHITE); 
        smallDisplay.print(litStr);
        
        smallDisplay.setFont(&Conthrax_SemiBold12pt7b); 
        smallDisplay.setCursor(refLitX + 6, fuelBaselineY); 
        smallDisplay.print("L");

        drawDebugBox(smallDisplay, refLitX - w50 - x50 - 2, fuelBaselineY + y50 - 2, w50 + x50 + w_lit_unit_max + 10, h50 + 4);
    }

    if (fabsf(displayInstKml - lastDispInstKml) >= 0.1f || fabsf(displayAvgKml - lastDispAvgKml) >= 0.1f || forceDraw) {
        lastDispInstKml = displayInstKml;
        lastDispAvgKml = displayAvgKml;

        // --- BLOCCO ISTANTRANEO (IST) ---
        int instCenterX = SMALL_CENTER_X + OFFSET_SMALL_KML_INST_X;
        int instY = SMALL_CENTER_Y + OFFSET_SMALL_KML_INST_Y;

        static int instBlockWidthMax = 0;
        static uint16_t badgeW_ist = 0, badgeH_ist = 0;
        static int16_t badgeBx_ist = 0, badgeBy_ist = 0;
        static uint16_t w_kml_unit = 0;
        static uint16_t w_num_max = 0;
        static bool instLayoutInit = false;

        if (!instLayoutInit) {
            instLayoutInit = true;
            int16_t tx1, ty1; uint16_t th1;

            smallDisplay.setFont(&DS_DIGIT15pt7b);
            smallDisplay.getTextBounds("00.0", 0, 0, &tx1, &ty1, &w_num_max, &th1);

            smallDisplay.setFont(&Conthrax_SemiBold4pt7b);
            smallDisplay.getTextBounds("KM/L", 0, 0, &tx1, &ty1, &w_kml_unit, &th1);

            uint16_t bw, bh;
            smallDisplay.getTextBounds("IST", 0, 0, &badgeBx_ist, &badgeBy_ist, &bw, &bh);
            badgeW_ist = bw + 6;
            badgeH_ist = bh + 4;

            uint16_t w_right = (badgeW_ist > w_kml_unit) ? badgeW_ist : w_kml_unit;
            instBlockWidthMax = w_num_max + 4 + w_right;
        }

        char instStr[8];
        if (displayInstKml > 0.0f) snprintf(instStr, sizeof(instStr), "%.1f", displayInstKml);
        else snprintf(instStr, sizeof(instStr), "0.0");

        int16_t tx15, ty15; uint16_t w_num, h_num;
        smallDisplay.setFont(&DS_DIGIT15pt7b);
        smallDisplay.getTextBounds(instStr, 0, 0, &tx15, &ty15, &w_num, &h_num);

        uint16_t w_right_ist = (badgeW_ist > w_kml_unit) ? badgeW_ist : w_kml_unit;
        int currentInstWidth = w_num + 4 + w_right_ist;
        int startInstX = instCenterX - (currentInstWidth / 2);

        int clearInstX = instCenterX - (instBlockWidthMax / 2) - 4;
        smallDisplay.fillRect(clearInstX, instY - 20, instBlockWidthMax + 8, 26, TFT_BLACK);

        smallDisplay.setFont(&DS_DIGIT15pt7b);
        smallDisplay.setCursor(startInstX - tx15, instY);
        smallDisplay.setTextColor(TFT_CYAN);
        smallDisplay.print(instStr);

        int rightColInstX = startInstX + w_num + 4;

        int instBadgeX = rightColInstX + (w_right_ist - badgeW_ist) / 2;
        int instBadgeY = instY - 18;
        drawAARoundRect(smallDisplay, instBadgeX, instBadgeY, badgeW_ist, badgeH_ist, 2, TFT_CYAN);
        smallDisplay.setFont(&Conthrax_SemiBold4pt7b);
        smallDisplay.setTextColor(TFT_CYAN);
        smallDisplay.setCursor(instBadgeX + 3 - badgeBx_ist, instBadgeY - badgeBy_ist + 2);
        smallDisplay.print("IST");

        int kmlInstX = rightColInstX + (w_right_ist - w_kml_unit) / 2;
        smallDisplay.setFont(&Conthrax_SemiBold4pt7b);
        smallDisplay.setTextColor(TFT_CYAN);
        smallDisplay.setCursor(kmlInstX, instY - 1);
        smallDisplay.print("KM/L");

        drawDebugBox(smallDisplay, startInstX - 2, instY - 20, currentInstWidth + 4, 24);

        // --- BLOCCO MEDIO (AVG) ---
        int avgCenterX = SMALL_CENTER_X + OFFSET_SMALL_KML_AVG_X;
        int avgY = SMALL_CENTER_Y + OFFSET_SMALL_KML_AVG_Y;

        static int avgBlockWidthMax = 0;
        static uint16_t badgeW_avg = 0, badgeH_avg = 0;
        static int16_t badgeBx_avg = 0, badgeBy_avg = 0;
        static bool avgLayoutInit = false;

        if (!avgLayoutInit) {
            avgLayoutInit = true;
            uint16_t bw, bh;

            smallDisplay.setFont(&Conthrax_SemiBold4pt7b);
            smallDisplay.getTextBounds("AVG", 0, 0, &badgeBx_avg, &badgeBy_avg, &bw, &bh);
            badgeW_avg = bw + 6;
            badgeH_avg = bh + 4;

            uint16_t w_right = (badgeW_avg > w_kml_unit) ? badgeW_avg : w_kml_unit;
            avgBlockWidthMax = w_num_max + 4 + w_right;
        }

        char avgStr[8];
        if (displayAvgKml > 0.0f) snprintf(avgStr, sizeof(avgStr), "%.1f", displayAvgKml);
        else snprintf(avgStr, sizeof(avgStr), "0.0");

        smallDisplay.setFont(&DS_DIGIT15pt7b);
        smallDisplay.getTextBounds(avgStr, 0, 0, &tx15, &ty15, &w_num, &h_num);

        uint16_t w_right_avg = (badgeW_avg > w_kml_unit) ? badgeW_avg : w_kml_unit;
        int currentAvgWidth = w_num + 4 + w_right_avg;
        int startAvgX = avgCenterX - (currentAvgWidth / 2);

        int clearAvgX = avgCenterX - (avgBlockWidthMax / 2) - 4;
        smallDisplay.fillRect(clearAvgX, avgY - 20, avgBlockWidthMax + 8, 26, TFT_BLACK);

        smallDisplay.setFont(&DS_DIGIT15pt7b);
        smallDisplay.setCursor(startAvgX - tx15, avgY);
        smallDisplay.setTextColor(TFT_YELLOW);
        smallDisplay.print(avgStr);

        int rightColAvgX = startAvgX + w_num + 4;

        int avgBadgeX = rightColAvgX + (w_right_avg - badgeW_avg) / 2;
        int avgBadgeY = avgY - 18;
        drawAARoundRect(smallDisplay, avgBadgeX, avgBadgeY, badgeW_avg, badgeH_avg, 2, TFT_YELLOW);
        smallDisplay.setFont(&Conthrax_SemiBold4pt7b);
        smallDisplay.setTextColor(TFT_YELLOW);
        smallDisplay.setCursor(avgBadgeX + 3 - badgeBx_avg, avgBadgeY - badgeBy_avg + 2);
        smallDisplay.print("AVG");

        int kmlAvgX = rightColAvgX + (w_right_avg - w_kml_unit) / 2;
        smallDisplay.setFont(&Conthrax_SemiBold4pt7b);
        smallDisplay.setTextColor(TFT_YELLOW);
        smallDisplay.setCursor(kmlAvgX, avgY - 1);
        smallDisplay.print("KM/L");

        drawDebugBox(smallDisplay, startAvgX - 2, avgY - 20, currentAvgWidth + 4, 24);
    }

    int currentTempInt = (int)displayTemp;
    if (currentTempInt != lastTempInt || forceDraw) {
        lastTempInt = currentTempInt;
        
        int tempX = SMALL_CENTER_X + OFFSET_SMALL_TEMP_X;
        int tempBottomY = SMALL_CENTER_Y + OFFSET_SMALL_TEMP_Y;

        static int refX = 0;
        static uint16_t w_num3_max = 0;
        static uint16_t w_unit_max = 0;
        static bool tempLayoutInit = false;

        if (!tempLayoutInit) {
            tempLayoutInit = true;
            int16_t tx1, ty1; uint16_t tw1, th1, tw2, th2;
            
            smallDisplay.setFont(&DS_DIGIT15pt7b);
            smallDisplay.getTextBounds("00", 0, 0, &tx1, &ty1, &tw1, &th1);

            smallDisplay.setFont(&Conthrax_SemiBold7pt7b);
            smallDisplay.getTextBounds("C", 0, 0, &tx1, &ty1, &tw2, &th2);
            w_unit_max = 5 + 2 + tw2;

            uint16_t total2W = tw1 + 3 + w_unit_max;
            int start2X = tempX - (total2W / 2);
            refX = start2X + tw1;

            smallDisplay.setFont(&DS_DIGIT15pt7b);
            smallDisplay.getTextBounds("188", 0, 0, &tx1, &ty1, &w_num3_max, &th1);
        }

        char tempNumStr[6]; snprintf(tempNumStr, sizeof(tempNumStr), "%d", currentTempInt);
        
        smallDisplay.setFont(&DS_DIGIT15pt7b); 
        smallDisplay.getTextBounds(tempNumStr, 0, 0, &x15, &y15, &w15, &h15);
        
        int tempBaselineY = (tempBottomY - h15) - y15; 

        int clearLeft = refX - w_num3_max - 4;
        int clearWidth = w_num3_max + 4 + w_unit_max + 8;
        smallDisplay.fillRect(clearLeft, tempBottomY - h15 - 2, clearWidth, h15 + 4, TFT_BLACK); 

        // --- ARCO TEMPERATURA (in Antialiasing a passo 0.5°) ---
        float tempFactor = constrain((displayTemp - 20.0f) / 100.0f, 0.0f, 1.0f);
        int targetTempIdx = (int)(tempFactor * (NUM_ARC_STEPS - 1));

        static int lastTempDrawnIdx = -1;
        static uint16_t lastTempDrawnColor = 0;

        uint16_t activeTempColor = TFT_GREEN;
        if (displayTemp > 0.1f && displayTemp < 50.0f) activeTempColor = TFT_CYAN;
        else if (displayTemp >= 95.0f) activeTempColor = TFT_RED;
        else if (displayTemp >= 85.0f) activeTempColor = TFT_YELLOW;
        else if (displayTemp <= 0.1f) activeTempColor = TFT_WHITE;

        bool colorChanged = (activeTempColor != lastTempDrawnColor);

        if (forceDraw || lastTempDrawnIdx < 0 || colorChanged) {
            for (int i = 0; i < NUM_ARC_STEPS; i++) {
                uint16_t segmentColor = (i <= targetTempIdx) ? activeTempColor : smallDisplay.color565(35, 35, 35);
                drawAALine(smallDisplay, tempArcLUT[i].xStart, tempArcLUT[i].yStart, tempArcLUT[i].xEnd, tempArcLUT[i].yEnd, segmentColor);
            }
            lastTempDrawnIdx = targetTempIdx;
            lastTempDrawnColor = activeTempColor;
        } else if (targetTempIdx != lastTempDrawnIdx) {
            if (targetTempIdx > lastTempDrawnIdx) {
                for (int i = lastTempDrawnIdx + 1; i <= targetTempIdx; i++) {
                    drawAALine(smallDisplay, tempArcLUT[i].xStart, tempArcLUT[i].yStart, tempArcLUT[i].xEnd, tempArcLUT[i].yEnd, activeTempColor);
                }
            } else {
                uint16_t offColor = smallDisplay.color565(35, 35, 35);
                for (int i = targetTempIdx + 1; i <= lastTempDrawnIdx; i++) {
                    drawAALine(smallDisplay, tempArcLUT[i].xStart, tempArcLUT[i].yStart, tempArcLUT[i].xEnd, tempArcLUT[i].yEnd, offColor);
                }
            }
            lastTempDrawnIdx = targetTempIdx;
        }

        smallDisplay.setFont(&DS_DIGIT15pt7b); 
        smallDisplay.setCursor(refX - w15 - x15, tempBaselineY);
        smallDisplay.setTextColor(activeTempColor); 
        smallDisplay.print(tempNumStr);
        
        int degCenterX = refX + 4;
        int degCenterY = tempBaselineY - h15 + 4;
        drawAACircle(smallDisplay, degCenterX, degCenterY, 2, activeTempColor);

        smallDisplay.setFont(&Conthrax_SemiBold7pt7b); 
        smallDisplay.setTextColor(activeTempColor);
        smallDisplay.setCursor(refX + 9, tempBaselineY);
        smallDisplay.print("C");

        drawDebugBox(smallDisplay, refX - w15 - x15 - 2, tempBaselineY + y15 - 2, w15 + x15 + w_unit_max + 8, h15 + 4);
    }

    smallDisplay.endWrite();
}

void checkNightMode(const SensorSnapshot& snap) {
    static bool isNightModeActive = false; 
    static bool firstCheck = true;
    
    if (snap.timeValid) {
        int localHour = snap.localHour;
        bool shouldBeNightMode = (localHour >= NIGHT_MODE_START_HOUR || localHour < NIGHT_MODE_END_HOUR);
        
        if (shouldBeNightMode != isNightModeActive || firstCheck) {
            isNightModeActive = shouldBeNightMode;
            firstCheck = false;
            
            uint8_t brightness = isNightModeActive ? 75 : 255;
            analogWrite(BL_BIG_DISPLAY, brightness); 
            analogWrite(BL_SMALL_DISPLAY, brightness);
        }
    }
}

void transmitTelemetry(const SensorSnapshot& snap) {
    char telemetryData[256];
    bool usingGps = snap.isGpsSpeedValid;
    
    snprintf(telemetryData, sizeof(telemetryData),
    "Src: %s | Circ: %.0fmm | Volume: %.2f L (%d%%) | Bat: %.1f V | Temp: %.1f C | Speed: %d KM/H | Sats: %d | TotalDist: %.2f KM | 0-50: %.2f S | FPS: %.1f (AVG: %.1f)",
    usingGps ? "GPS" : "HALL", WHEEL_CIRCUMFERENCE_MM, snap.fuelLiters, snap.fuelPercentage, snap.batteryVoltage, snap.engineTemperature,
    (int)snap.currentSpeed, snap.satellites, snap.totalDistanceKm, snap.accelResultTime, currentMeasuredFps, currentAverageFps);
    
    Serial.println(telemetryData);
}
